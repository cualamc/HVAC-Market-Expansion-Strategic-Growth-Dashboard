MEASURE '1_time&measures'[revenuetrend] = 
VAR SixMonthWindow = 
    DATESINPERIOD(
        '2_cardone_details'[date_invoice], 
        MAX('2_cardone_details'[date_invoice]), 
        -6, 
        MONTH
    )
VAR MonthlyData = 
    FILTER(
        SUMMARIZE(
            CALCULATETABLE('2_cardone_details', SixMonthWindow),
            '2_cardone_details'[date_invoice],
            "MonthlyRev", [rev_sum]
        ),
        [MonthlyRev] > 0
    )
VAR StartRev = MINX(TOPN(1, MonthlyData, '2_cardone_details'[date_invoice], ASC), [MonthlyRev])
VAR EndRev = MINX(TOPN(1, MonthlyData, '2_cardone_details'[date_invoice], DESC), [MonthlyRev])
VAR MonthCount = COUNTROWS(MonthlyData)

RETURN
IF(MonthCount < 2, "New/Intermittent",
    VAR Growth = DIVIDE(EndRev - StartRev, StartRev)
    RETURN
    SWITCH(TRUE(),
        Growth > 0.05, "Up ▲",
        Growth < -0.05, "Down ▼",
        "Steady ▬"
    )
)
MEASURE '1_time&measures'[revenuetrend_score] = 
VAR SixMonthWindow = DATESINPERIOD('2_cardone_details'[date_invoice], MAX('2_cardone_details'[date_invoice]), -6, MONTH)

VAR MonthlyData = 
    FILTER(
        SUMMARIZE(
            CALCULATETABLE(
                '2_cardone_details', 
                SixMonthWindow,
                '3_grouped_zip'[cardone_details.check_existingarea]="yes"
            ),
            '2_cardone_details'[date_invoice],
            "MonthlyRev", [rev_sum]
        ),
        [MonthlyRev] > 0
    )

VAR StartRev = MINX(TOPN(1, MonthlyData, '2_cardone_details'[date_invoice], ASC), [MonthlyRev])
VAR EndRev = MINX(TOPN(1, MonthlyData, '2_cardone_details'[date_invoice], DESC), [MonthlyRev])
VAR MonthCount = COUNTROWS(MonthlyData)

RETURN
IF(MonthCount < 2, 50, -- Default to neutral for new markets
    VAR GrowthRate = DIVIDE(EndRev - StartRev, StartRev)
    
    -- Normalization Logic: 
    -- 0% growth = 50 points
    -- +20% growth = 100 points
    -- -20% growth = 0 points
    VAR NormalizedScore = 50 + (GrowthRate * 250) 
    
    RETURN
    MIN(MAX(NormalizedScore, 0), 100) -- Ensures score stays between 0 and 100
)
